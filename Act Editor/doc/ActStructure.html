Act file structure

This document presents the structure of the Act object. All the elements presented below can be modified directly from a script. However, you must use a Backup command before doing so.

Act

public class Act : IEnumerable<Action>, IDisposable {
	public IActConverter ActConverter;	// Gets or sets the converter to save (or read) the Act file
	public ActHeader Header;	// Gets the ActHeader object which contains information regarding the file format version
 
	public string LoadedPath;	// Gets the path of the loaded file (might be null)
	public Spr Sprite;		// Gets the sprite object attahed to the Act
	public string Name;		// Gets the name of the Act (this is used by references)
	public List<string> SoundFiles;	// Gets the sound list
	public int NumberOfActions;	// Gets the number of actions
 
	public Action this[int index]; // Returns the action
	public Frame this[int actionIndex, int frameIndex]; // Returns the frame
	public Layer this[int actionIndex, int frameIndex, int layerIndex]; // Returns the layer

	//  Returns a frame or null if it is not found
	public Frame TryGetFrame(int actionIndex, int frameIndex);
 
	// Returns a layer or null if it is not found
	public Layer TryGetLayer(int actionIndex, int frameIndex, int layerIndex);
 
	// Executes an action on all layers
	public void AllLayers(Action<Layer> func);
 
	// Returns a list with all the frames
	public List<Frame> GetAllFrames();
 
	// Returns a list with all the layers
	public List<Layer> GetAllLayers();
 
	// Update Act Editor's visual controls
	public void InvalidateVisual();
}

Action

public class Action {
	public int NumberOfFrames;		// Gets the number of frames
	public List<Frame> Frames;		// List of all frames
 
	public float AnimationSpeed = 6;	// Gets or sets the speed of the animation (must be multiplied by 25 for the interval in ms)
	public int Interval;	// Gets the interval between two frames in milliseconds
	
	public Frame this[int frameIndex];	// Returns the frame
	public Layer this[int frameIndex, int layerIndex];	// Returns the layer
}

Frame

public class Frame {
	public int NumberOfLayers;		// Gets the number of layers
	public int SoundId;			// Gets or sets the sound id
	public List<Layer> Layers;	// List of all layers
	public List<Anchor> Anchors;		// List of all anchors
 
	public Layer this[int index];	// Returns the layer
}

Layer

public class Layer {
	public GrfColor Color;	// Gets or sets the color mask
 
	public int OffsetX;	// Gets or sets the offset in X
	public int OffsetY;	// Gets or sets the offset in Y
	public float ScaleX;	// Gets or sets the scale in X
	public float ScaleY;	// Gets or sets the scale in Y
	public int Rotation;	// Gets or sets the rotation
	public bool Mirror;	// Gets or sets the mirror state (if true, this is the equivalent of -1 * ScaleX)
 
	public int SpriteIndex;		// Gets or sets the relative sprite index (for Bgra32 images, you must substract the number of Indexed8 images)
	public SpriteTypes SpriteType;	// Gets or sets the sprite type
	public int SpriteTypeInt;	// Gets or sets the sprite type (int based; 0 = Indexed8, 1 = Bgra32)
 
	public int Height;	// Gets the height of the current image (this property is overwritten when saving the Act)
	public int Width;	// Gets the width of the current image (this property is overwritten when saving the Act)
}
